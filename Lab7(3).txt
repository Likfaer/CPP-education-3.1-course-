#include <iostream>
#include <string.h>
#include <windows.h>
using namespace std;

template <class Type, int MAXSIZE>
class Stack {
	Type* Pointer;
	int top;
public:
	Stack ()
	{
		Pointer = new Type[MAXSIZE];
		top = 0;
	}
	void push(Type CC)
	{
		if (top == MAXSIZE) {
			cout << "Стэк переполнен [" << top << "/"<< MAXSIZE<< "]" << endl;
			return;
		}
		Pointer[top] = CC;
		top++;
	}
	void pop(Type)
	{
		if (top == 0) {
			cout << "Стэк пуст [" << top << "/" << MAXSIZE << "]" << endl;
			return ;
		}
		Pointer[top-1] = 0;
		top--;
	}
	void show()
	{
		for (int i = 0; i < MAXSIZE; i++)
			cout << i << " : " << Pointer[i] << endl;
	}
};
int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	int i;
	Stack <int, 10> A;
	for (i = 0; i < 10; i++)
		A.push(i + 25);
	A.show();
	A.pop();
	A.show();
	Stack <double,100> B;
}
